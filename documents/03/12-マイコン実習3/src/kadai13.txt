#include <p30F3013.h>

void wait1() {
  unsigned i, j;
  for (i = 0; i < 64; i++) {
    for (j = 0; j < 65000; j++) {
      asm("nop");
    }
  }
}

void main() {
  TRISB = 0;
  int i = 0;

  while (1) {
    LATB = 1;
    wait1();
    for (i = 0; i < 7; i++) {
      LATB = LATB << 1;
      wait1();
    }
  }
}
